G6 DEMO - 起步
  1. 定义数据源，声明节点、边
  2. 创建画布实例
  3. 设置数据
  4. 渲染画布
        
          // 1.定义数据源，声明节点、边
          const data = {
            nodes: [
              {
                id: 'node001',
                label: 'node001',
                x: 100,
                y: 200,
                shape: 'rect'
              },
              {
                id: 'node002',
                label: 'node002',
                x: 300,
                y: 200,
                shape: 'circle'
              }
            ],
            edges: [
              // 指定边的起点、终点id
              {
                source: 'node001',
                target: 'node002'
              }
            ]
          }
          // 2.创建画布实例
          const graph = new G6.Graph({
            // 容器ID
            container: 'container',
            width: 500,
            height: 400
          })
          // 3.设置数据
          graph.data(data)
          // 4.渲染画布
          graph.render()
